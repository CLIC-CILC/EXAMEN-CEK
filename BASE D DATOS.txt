CREATE TABLE VENGADOR(
ID_VENGADOR INT UNSIGNED NOT NULL AUTO_INCREMENT,
NOM_VENGADOR VARCHAR(500) NOT NULL,
F_CAPTURA DATETIME,
PRIMARY KEY(ID_VENGADOR));

CREATE TABLE EDITOR(
ID_EDITOR INT UNSIGNED NOT NULL AUTO_INCREMENT,
NOM_EDITOR VARCHAR(500) NOT NULL,
F_CAPTURA DATETIME,
PRIMARY KEY(ID_EDITOR));

CREATE TABLE ESCRITOR(
ID_ESCRITOR INT UNSIGNED NOT NULL AUTO_INCREMENT,
NOM_ESCRITOR VARCHAR(500) NOT NULL,
F_CAPTURA DATETIME,
PRIMARY KEY(ID_ESCRITOR));

CREATE TABLE COLORISTA(
ID_COLORISTA INT UNSIGNED NOT NULL AUTO_INCREMENT,
NOM_COLORISTAR VARCHAR(500) NOT NULL,
F_CAPTURA DATETIME,
PRIMARY KEY(ID_COLORISTA));

CREATE TABLE COMIC(
ID_COMIC INT UNSIGNED NOT NULL AUTO_INCREMENT,
NOM_COMIC VARCHAR(500) NOT NULL,
F_CAPTURA DATETIME,
PRIMARY KEY(ID_COMIC));

CREATE TABLE BIBLIOTECA(
ID_VENGADOR INT UNSIGNED NOT NULL,
ID_EDITOR INT UNSIGNED NOT NULL,
ID_ESCRITOR INT UNSIGNED NOT NULL,
ID_COLORISTA INT UNSIGNED NOT NULL,
ID_COMIC INT UNSIGNED NOT NULL,
CVE_CAPTURA INT UNSIGNED NOT NULL,
F_CAPTURA DATETIME NOT NULL,
PRIMARY KEY (ID_VENGADOR, ID_EDITOR, ID_ESCRITOR, ID_COLORISTA, ID_COMIC),
CONSTRAINT 'BIBLIOTECA_FK_1' FOREIGN KEY(ID_VENGADOR) REFERENCES VENGADOR(ID_VENGADOR),
CONSTRAINT 'BIBLIOTECA_FK_2' FOREIGN KEY(ID_EDITOR) REFERENCES EDITOR(ID_EDITOR),
CONSTRAINT 'BIBLIOTECA_FK_3' FOREIGN KEY(ID_ESCRITOR) REFERENCES ESCRITOR(ID_ESCRITOR),
CONSTRAINT 'BIBLIOTECA_FK_4' FOREIGN KEY(ID_COLORISTA) REFERENCES COLORISTA(ID_COLORISTA),
CONSTRAINT 'BIBLIOTECA_FK_4' FOREIGN KEY(ID_COMIC) REFERENCES COMIC(ID_COMIC));
